import importlib
import os
import sys
import types
import typing

import click
from flask import Flask, Response, request
from utils.core import exit_process
from utils.decorators import with_debug
from utils.logging import log_exception, log_message
from utils.workspace import wtp_workspace_root


@with_debug
def run_wtp_cli(debug: bool, port: int) -> None:
    """runs the wtp_cli project with optional debugging"""
    workspace_root = wtp_workspace_root(os.getcwd())

    os.chdir(workspace_root)

    cli_module: types.ModuleType = importlib.import_module(
        "projects.wtp_cli.wtp_cli.cli"
    )

    if os.environ.get("WTP_DEBUG") is not None:
        # TODO: find a better way of doing this - it's kinda sketchy
        old_argv = [*sys.argv]
        new_argv = [old_argv[0], *old_argv[5::]]

        sys.argv = new_argv

        cli_module.cli()


def start_dev_server(handler: str, port: int, debug: bool = True) -> None:
    app = Flask(__name__)
    pid = os.getpid()
    pkg: types.ModuleType

    log_message("Starting...")
    log_message(f"-  {handler}")

    log_message(f"Debug process pid - {pid}")

    lambda_name = handler.split(".")[-1]

    try:
        pkg = importlib.import_module(handler)
    except ModuleNotFoundError as ex:
        log_exception(ex, message=f"Module {handler} not found")
        exit_process(should_log=True)

        exit(1)

    # TODO: move this to try/catch
    if not hasattr(pkg, "lambda_handler"):
        log_message(
            f"module '{handler}' does not have a lambda handler called 'lambda_handler'",
            level="error",
        )
        exit_process(should_log=True)

        exit(1)

    @app.route(f"/lambdas/{lambda_name}", methods=["POST"])
    def post() -> Response:
        log_message(f"handler - {handler}")

        data: typing.Dict[str, object] = {}
        content_type = request.headers.get("Content-Type")

        if content_type == "application/json":
            data = typing.cast(typing.Dict[str, object], request.json)
        else:
            return Response({"body": "Content-Type not supported", "code": 400})

        result = pkg.lambda_handler(data["event"], data["context"])

        return result

    app.run(
        debug=debug,
        port=port,
        use_reloader=False if debug else True,
    )


# cli configuration
@click.group(help="Run a project on a local dev server")
def run() -> None:
    pass


@click.command(help="Run an AWS lambda, isolated to the local environment")
@click.option(
    "--port", "-p", type=int, default=5678, help="port used to accept requests"
)
@click.option(
    "--debug",
    "-d",
    type=bool,
    default=False,
    is_flag=True,
    help="expose a debug port or not",
)
@click.argument("handler")
def aws_lambda(handler: str, port: int, debug: bool) -> None:
    start_dev_server(handler=handler, debug=debug, port=port)


@click.command(
    name="wtp_cli", help="Run an AWS lambda, isolated to the local environment"
)
@click.option(
    "--port", "-p", type=int, default=5678, help="port used to accept requests"
)
@click.option(
    "--debug",
    "-d",
    type=bool,
    default=False,
    is_flag=True,
    help="expose a debug port or not",
)
@click.argument("command", nargs=-1)
def wtp_cli(command: str, port: int, debug: bool) -> None:
    run_wtp_cli(debug=debug, port=port)


run.add_command(aws_lambda)
run.add_command(wtp_cli)
